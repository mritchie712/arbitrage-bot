def getBTCE():

    #https://btc-e.com/api/3/depth/btc_usd
    btceInfoURL = btcecoreURL + 'info'

            print str(e)

        tempDF['BidAsk']='bids'

        symImOn = sym.keys()[0]
        tempDF = pd.DataFrame(btceBookLS[i][symImOn]['asks'])
        tempDF['symbol']=symImOn
        tempDF['BidAsk']='asks'
        btceBook=btceBook.append(tempDF)
        i+=1

def getStmp():
    coreURL = 'https://www.bitstamp.net/api/'
    ############Finx BTC###########################
    tranURL = coreURL + 'transactions/'
    tranReq = urllib2.Request(tranURL, headers={'User-Agent' : "Magic Browser"}) 
    tranRes = urllib2.urlopen(tranReq)
    stmpTran = json.loads(tranRes.read())
    stmpTranDF = pd.DataFrame(stmpTran)
    stmpTranMaxDate = stmpTranDF.date.max()
    public_client = bitstamp.client.Public()
    stmpTickerDict = public_client.ticker()
    stmpTick = pd.DataFrame(stmpTickerDict,index=[0])
    stmpTick['crypto'] = 'BTC'
    stmpTick['fiat'] = 'USD'
    stmpTick['server_time'] = (datetime.datetime.fromtimestamp(int(stmpTick['timestamp'])).strftime('%Y-%m-%d %H:%M:%S'))
    stmpDepthDict = public_client.order_book()
    stmpDepthAsk = pd.DataFrame(stmpDepthDict['asks'],columns = ['price','amount'])
    stmpDepthAsk['BidAsk'] = 'asks'
    stmpDepthBid = pd.DataFrame(stmpDepthDict['bids'],columns = ['price','amount'])
    stmpDepthBid['BidAsk'] = 'bids'
    stmpBook = stmpDepthAsk.append(stmpDepthBid)
    stmpBook['market'] = 'stmp'
    stmpBook['timestamp'] = stmpTranMaxDate
    stmpBook['symbol'] = 'btcusd'
    return stmpBook

def getFinx(finxLmtIn):
    coreURL = 'https://api.bitfinex.com/v1/'
    symURL = coreURL + 'symbols'
    symReq = urllib2.Request(symURL, headers={'User-Agent' : "Magic Browser"}) 
    symRes = urllib2.urlopen(symReq)
    finxSym = json.loads(symRes.read())
    bookURL = coreURL + 'book/'
    finxLmt = finxLmtIn
    finxLmtURL = '/?limit_bids=' + str(finxLmt) + '&limit_asks=' + str(finxLmt)
    bookLS=[]
    for sym in finxSym:
        request = urllib2.Request(bookURL+sym+finxLmtURL, headers={'User-Agent' : "Magic Browser"}) 
        response = urllib2.urlopen(request)
        finxBookJSON = json.loads(response.read())
        ls = [sym,finxBookJSON]
        bookLS.append(ls)
    #finxBook = pd.DataFrame(bookLS,index=[0])
    finxBook = pd.DataFrame()
    i = 0
    while i < len(bookLS):
        tempDF = pd.DataFrame(bookLS[i][1]['bids'])
        tempDF['symbol']=bookLS[i][0]
        tempDF['BidAsk']='bids'
        finxBook=finxBook.append(tempDF)
        i+=1
    i = 0
    while i < len(bookLS):
        tempDF = pd.DataFrame(bookLS[i][1]['asks'])
        tempDF['symbol']=bookLS[i][0]
        tempDF['BidAsk']='asks'
        finxBook=finxBook.append(tempDF)
        i+=1
    #finxBook.symbol.value_counts()
    finxBook['market'] = 'finx'
    return finxBook

            'arbPrcntCt':arbPrcntCt,
            'arbInd':arbInd}

        except Exception, e:
            print str(e)
